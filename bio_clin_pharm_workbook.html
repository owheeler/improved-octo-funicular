1. Get Data

library(haven)
library(DT)
adae <- read_sas("https://github.com/elong0527/r4csr/raw/master/data-adam/adae.sas7bdat", 
    NULL)

datatable(adae)

2. Visualize the Data

library(ggplot2)
ggplot(adae) +
  aes(x = AESLIFE, y = AGE, fill = RACE) +
  geom_violin(adjust = 1L, scale = "area") +
  scale_fill_hue(direction = 1) +
  theme_minimal()
